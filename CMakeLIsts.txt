cmake_minimum_required(VERSION 3.16)

project(ActivityManager
    VERSION 0.1
    LANGUAGES CXX
)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

qt_standard_project_setup()

qt_add_executable(ActivityManager
    main.cpp
    model/AgendaUniversitaria.h
    model/Priority.h
    model/Lezione.h
    model/Esame.h
    model/Location.h
    model/Consegna.h
    model/Ricevimento.h
    model/Lezione.cpp
    model/Consegna.cpp
    model/Ricevimento.cpp
)

target_include_directories(ActivityManager
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_link_libraries(ActivityManager
    PRIVATE
        Qt6::Widgets
)

set_target_properties(ActivityManager PROPERTIES
    WIN32_EXECUTABLE TRUE
)
